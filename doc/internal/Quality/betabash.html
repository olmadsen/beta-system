Bash aliases for using BETA

# CVS setup
alias cvs-setup='export CVS_RSH="ssh"; export CVSROOT="beta@alex:/users/beta/.CVSHOME"'

# RCM setup:
#   - I put rcm.exe in C:/bin. Here added to path.
#   - Assumes \\dodeca\beta mapped to drive F.
#   - rcm requires TCL8.0. But 8.3 (as here) will do,
#     if tcl83.dll is copied to tcl80.dll
#   - rcm requires rcs. I use the one in the WinCVS package.
#   - Make sure PATH is *appended* to, since the cvs in WinCVS is 
#     older than the one in cygwin
#   - Make sure /tmp exists
alias rcm-setup='export USER="datpete"; export RAGNAROOT="F:/.Ragnarok"; \
      export TCL_LIBRARY="C:\tcl\lib\tcl8.3"; \
      export PATH="${PATH}:/cygdrive/c/tcl/bin:/cygdrive/c/bin:/cygdrive/c/Program Files/WinCVS"'
alias rcm='pushd $BETALIB/compiler; \rcm; popd'
alias rcmup='pushd $BETALIB/compiler; \rcm -do "gettip; quit"; popd'

# MS Visual Studio 6.0 setup
#    - Setting this up using an alias instead of using
#      Control Panels->Advanced->Environment allows Visual.NET to be
#      installed and MSVS6 only to take precedence when the alias is used.
alias msdev-setup='export INCLUDE="c:/Program Files/Microsoft Visual Studio/VC98/Include/;$INCLUDE"; \
      export PATH="/cygdrive/c/Program Files/Microsoft Visual Studio/VC98/Bin/:\
                   /cygdrive/c/Program Files/Microsoft Visual Studio/Common/MSDev98/Bin/:$PATH"; \
      export LIB="c:/Program Files/Microsoft Visual Studio/VC98/Lib/;$LIB"'

# BETA Setup
#    - Assumes BETALIB set in advance (see aliases below)
#    - PATH must include path to wrapper scripts (CVS repository
#      bin_ntiwrappers)
#    - Setting OLD_BETALIB to the same as BETALIB make bootbeta use 
#      beta itself to boot
alias beta-setup='cvs-setup; export MIASDK="ms"; \
      export PATH="/cygdrive/c/beta/bin:/cygdrive/c/beta/bin/admin:$PATH"; \
      export OLD_BETALIB=$BETALIB; \
      cd $BETALIB'

alias r5.3='export BETALIB="c:\beta\r5.3"; beta-setup; msdev-setup; rcm-setup; PS1="[r5.3-$MIASDK] $PS1"'

alias r5.2.2='export BETALIB="c:\beta\r5.2.2"; beta-setup; msdev-setup; rcm-setup; PS1="[r5.2.2-$MIASDK] $PS1"'
