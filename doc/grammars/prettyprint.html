PRETTYPRINT Grammar

-- prettyprint : Agrammar : metagrammar -- 
Grammar prettyprint:


option

   suffix='.pgram'
   bobsoptions = '25, 32, 34'
   combegin    = '(*'
   comEnd      = '*)'
   stringChar  = '\''
   

rule

   
   <PrettyPrint>   ::= 'PrettyPrintScheme' 
                       'for'  ':' <ProductionList>;
   <ProductionList>::* <Production> ';' ;
   
   <Production>    ::| <Constructor> | <ListProd> ;
   <Constructor>   ::=  '=' ;
   <ListProd>      ::=  '=' '(' <ListSpec> ')';
   
   <ItemList>      ::* <Item>;
   <Item>          ::| <Terminal> | <NonTerm> | <Break> | <Block>
		     | <CommentPlace>;
   
   <Terminal>      ::| <DefaultTerm> | <AltTerm> ;
   <DefaultTerm>   ::= 'T' ':' ;
   <AltTerm>       ::=  ;
   
   <NonTerm>       ::= 'N' ':' ;
   
   <Break>         ::| <DefaultBreak> | <AltBreak> ;
   <DefaultBreak>  ::= '$$';
   <AltBreak>      ::= '$'  ',' ;
   
   <Block>         ::= '[' <BlockType> <ItemList> ']';

   (* comments must only be specified after terminals! *)
   <CommentPlace>  ::= '*';
   
   <ListSpec>      ::= 
                       '{' <BlockType>  '}'
		        ;
					   
   <BlockType>     ::| <Consistent> | <InConsistent> ;
   <Consistent>    ::= 'c';
   <InConsistent>  ::= 'i'
   

Attribute

   <Constructor> : 1
   <ListProd>    : 1
   <DefaultTerm> : 2
   <AltTerm>     : 2
   <NonTerm>     : 2
   <ListSpec>    : 2
